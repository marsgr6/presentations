Projectile motion¶

  • https://en.wikipedia.org/wiki/Projectile_motion

  • Displacement

    • At any time $t$, the projectile's horizontal and vertical displacement are:
$$ x = v_{0x} t = v_0 t cos(\theta) $$$$ y = v_{0y} t - \frac{1}{2} g t^2 = v_{0} t sin(\theta) - \frac{1}{2} g t^2 $$
  • Time of flight or total time of the whole journey
$$T = \frac{2 v_0 sin(\theta)}{g}$$

import numpy as np
import matplotlib.pyplot as plt
from scipy.stats import norm
from ipywidgets import interact, interactive, fixed, interact_manual
import ipywidgets as widgets

def plot_projectiles(theta):
    #initialize variables
    #velocity, gravity
    v = 30 
    g = 9.81  #improved g to standard precision, set it to positive
    #increment theta 25 to 90 then find  t, x, y
    #define x and y as arrays
    theta_rad = np.array([theta, 45])/180.0*np.pi #convert to radians, watch out for modulo division
    print("Theta:", np.array([theta, 45]))
    plt.figure()

    tmax = ((2 * v) * np.sin(theta_rad)) / g
    timemat = tmax*np.linspace(0,1,100)[:,None] #create time vectors for each angle

    x = ((v * timemat) * np.cos(theta_rad))
    y = ((v * timemat) * np.sin(theta_rad)) - ((0.5 * g) * (timemat ** 2))

    plt.figure(figsize=(20,10))
    plt.plot(x,y) #plot each dataset: columns of x and columns of y
    plt.ylim([0,50])
    plt.legend(np.array([theta, 45]))
    plt.show()

def throw_darts(darts, media1, media2, s1, s2):
    dart_x = norm.rvs(media1, s1, size=darts)
    dart_y = norm.rvs(media2, s2, size=darts)

    plt.figure(figsize=(10,10))

    plt.plot(0,0,'ok')
    plt.plot(dart_x, dart_y, 'x')

    circle1 = plt.Circle((0,0),1,color='b', fill=False)
    plt.gcf().gca().add_artist(circle1)
    circle1 = plt.Circle((0,0),2,color='g', fill=False)
    plt.gcf().gca().add_artist(circle1)
    circle1 = plt.Circle((0,0),3,color='r', fill=False)
    plt.gcf().gca().add_artist(circle1)

    plt.xlim(-5, 5)
    plt.ylim(-5, 5)
    plt.gca().set_aspect('equal', adjustable='box')
